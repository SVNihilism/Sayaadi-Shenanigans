	}

elder_culture_undead_suicide_effect = { 
	if = {
		limit = {
			any_vassal = {
				count >= 1
			}
		}
		every_vassal = {
			custom = all_vassals
			add_opinion = {
				modifier = suicide_opinion
				target = root
				opinion = -15
			}
		}
	}
	dynasty = { add_dynasty_prestige = medium_dynasty_prestige_loss }
	add_piety = major_piety_loss
	override_death_effect = { death_reason = death_vanished } 
}
