## Standard effect for host

	# Renown
	if = {
		limit = {
			culture = { has_cultural_parameter = renown_from_feasts }
			exists = dynasty
		}
		dynasty = { add_dynasty_prestige = minor_dynasty_prestige_value }
	}

	# Piety gain
	if = {
		limit = { 
			faith = {
				has_doctrine_parameter = wcss_harem_piety_gain_or_loss
			}
		}
		add_piety = {
			value = medium_piety_value
			if = {
				limit = {
					has_royal_court = yes
					wcss_lodging_fashion_value >= 9
				}
				multiply = 2.5
			}
			else_if = {
				limit = {
					has_royal_court = yes
					wcss_lodging_fashion_value >= 7
				}
				multiply = 2
			}
			else_if = {
				limit = {
					has_royal_court = yes
					wcss_lodging_fashion_value >= 5
				}
				multiply = 1.5
			}
			else_if = {
				limit = {
					has_royal_court = yes
					wcss_lodging_fashion_value <= 2
				}
				multiply = 0.5
			}
		}
	}

	# County development
	scope:activity_location.county = {
		add_county_modifier = {
			modifier = wcss_harem_recent_debauch_modifier
			years = 5
		}
		if = {

				scope:activity.activity_location.culture = { has_cultural_parameter = wcss_hedonist_bonus } # Hedonist bonus
			}
