wcss_virtuous_harem = { # Virtuous orgy - carnal faith
	icon = wcss_faith
	is_valid = {
		is_current_phase_active = yes
		any_attending_character = {
			wcss_harem_pulse_action_trigger = yes
			location = scope:province
			faith = { trait_is_virtue = lustful }
		}
	}

	weight = {
		value = 1
	}

	effect = {		
		random_attending_character = {
			limit = {
				wcss_harem_pulse_action_trigger = yes
				location = scope:province
				faith = { trait_is_virtue = lustful }
			}
			save_scope_as = first
		}
		
		add_activity_log_entry = {
			key = wcss_virtuous_harem
			tags = { pulse_action }
			character = scope:first
			
			scope:first = {
				add_piety = minor_piety_gain
			}
		}
	}
}

wcss_dancers_entertain_guests = { # Guests are entertained by dancers
	icon = positive
	is_valid = {
		is_current_phase_active = yes
		has_activity_type = wcss_activity_harem
		NOT = {
			has_activity_option = {
				category = wcss_harem_option_entertainment
				option = wcss_harem_entertainment_bad
			}
		}
		any_attending_character = {
			wcss_harem_pulse_action_trigger = yes
			location = scope:province
			likes_character_trigger = { CHARACTER = scope:host }
		}
	}

	weight = {
		value = 1
		if = {
			limit = {
				has_activity_type = wcss_activity_harem
				has_activity_option = {
					category = wcss_harem_option_entertainment
					option = wcss_harem_entertainment_normal
				}
			}
			add = -0.5
		}
	}

	effect = {			
		add_activity_log_entry = {
			key = wcss_dancers_entertain_guests
			tags = { pulse_action }
			character = scope:host

			if = {
				limit = {
					has_activity_type = wcss_activity_harem
					has_activity_option = {
						category = wcss_harem_option_entertainment
						option = wcss_harem_entertainment_good
					}						
				}
				scope:host = { add_prestige = medium_prestige_gain }					
			}
			else = {
				scope:host = { add_prestige = minor_prestige_gain }
			}			
			every_attending_character = {
				limit = { 
					wcss_harem_pulse_action_trigger = yes
					location = scope:province
					likes_character_trigger = { CHARACTER = scope:host }
				}
				custom = wcss_every_attendee
				if = {
					limit = {
						root = {
							has_activity_type = wcss_activity_harem
							has_activity_option = {
								category = wcss_harem_option_entertainment
								option = wcss_harem_entertainment_good
							}
						}						
					}
					add_opinion = {
						modifier = impressed_opinion
						target = scope:host
						opinion = 20
					}						
				}
				else = {
					add_opinion = {
						modifier = impressed_opinion
						target = scope:host
						opinion = 10
					}
				}
			}
		}
	}
}

wcss_guests_charm_others = { # Charming guest
	icon = lover
	is_valid = {
		is_current_phase_active = yes
		any_attending_character = {
			count >= 2
			wcss_harem_pulse_action_trigger = yes
			location = scope:province
		}
	}

	weight = 1

	effect = {
		random_attending_character = {
			limit = {
				wcss_harem_pulse_action_trigger = yes
				location = scope:province
			}
			weight = {
				base = 1
				modifier = {
					add = 10
					has_personality_extroverted_trigger = yes
				}
				modifier = {
					factor = 5
					OR = {
						has_trait = seducer
						has_focus = intrigue_temptation_focus
					}
				}
			}
			save_scope_as = first
		}
		random_attending_character = {
			limit = {
				wcss_harem_pulse_action_trigger = yes
				location = scope:province
				NOT = { this = scope:first }
			}
			weight = {
				base = 1
				modifier = {
					add = 10
					has_personality_introverted_trigger = no
				}
				modifier = {
					factor = 0.1
					NOT = { 
						scope:first = { 
							likes_character_trigger = { CHARACTER = prev } 
						}
					}
				}
			}
			save_scope_as = second
		}
		add_activity_log_entry = {
			key = wcss_guests_charm_others
			tags = { pulse_action }
			character = scope:first

			scope:second = {
				add_opinion = {
					modifier = romance_opinion
					target = scope:first
					opinion = 20
				}
				hidden_effect = {
					if = {
						limit = {
							can_set_relation_potential_lover_trigger = { CHARACTER = scope:first }
						}
						set_relation_potential_lover = scope:first
					}
				}
			}
		}
	}
}
